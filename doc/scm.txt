# 1. Create and activate virtual environment
sudo apt-get install python3-venv    # Install venv if not available
python3 -m venv .venv
source .venv/bin/activate

## 2. Add Microsoft repository for SQL Server (optional, for DB connectivity)
## curl https://packages.microsoft.com/keys/microsoft.asc | sudo apt-key add -
## sudo apt-get update

## 3. Install ODBC Driver for SQL Server (only if connecting to MS SQL)
## sudo ACCEPT_EULA=Y apt-get install msodbcsql17

# 2. Descargar e instalar la clave y el repositorio de Microsoft para Ubuntu/Debian
curl https://packages.microsoft.com/keys/microsoft.asc | sudo tee /etc/apt/trusted.gpg.4.d/microsoft.asc

# Añadir el repositorio correcto
curl https://packages.microsoft.com/config/ubuntu/$(lsb_release -rs)/prod.list | sudo tee /etc/apt/sources.list.d/mssql-release.list

# 2. Actualizar los repositorios del sistema
sudo apt-get update

# 3. Instalar el ODBC Driver 18 (y unixodbc)
# Te pedirá aceptar los términos de licencia en la terminal (puedes usar las flechas y Enter)
sudo ACCEPT_EULA=Y apt-get install -y msodbcsql18 unixodbc-dev


# 4. Install Python packages
pip install django djangorestframework pyodbc mssql-django django-cors-headers

# 5. ✅ ADD TENSORFLOW FOR AI MODEL TRAINING
pip install tensorflow

# 6. Optional shorcut for both 4. and 5. 
pip freeze > requirements.txt
# conflicting versions delete version number
pip install -r requirements.txt

# 7. Run server 
python manage.py runserver
